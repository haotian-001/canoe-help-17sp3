Examples
This example shows how to read the Dtcs and print a small report.
Copy
public void PrintDtcReportScript()
{
  // Use the ECU currently selected by the Application
  Ecu myEcu = Application.GetEcu();

  ReadDtcResult result = myEcu.ReadDtcs();
  if (result.Status != FaultMemoryStatus.Success)
  {
    // Handle error ...
  }

  foreach (Dtc dtc in result.Dtcs)
  {
    // Print out DTC information. e.g. "P0010 11 Camshaft Position Actuator Circuit short to ground: 0x09"
    Output.WriteLine(dtc.SaeCode + " - " + dtc.ErrorText + ": 0x" + dtc.Status.ToString("x2"));
  }
}
