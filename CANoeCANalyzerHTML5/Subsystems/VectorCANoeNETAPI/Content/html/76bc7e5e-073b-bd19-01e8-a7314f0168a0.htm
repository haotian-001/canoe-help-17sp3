 9  // Configure and open a connection

12
13  // Register the event handlers
14  supply.OnSerialSentEvent += this.SendHandler;

16  supply.OnSerialErrorEvent += this.ErrorHandler;

18  // Send a string

20
21  // Receive data
22  var receiveBuffer = new byte[32];
23  supply.SerialReceive(receiveBuffer);
