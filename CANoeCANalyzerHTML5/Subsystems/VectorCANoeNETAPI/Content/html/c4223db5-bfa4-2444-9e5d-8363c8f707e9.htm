Remarks
The method marked in this way must be a non-static public method, take a TransientCallContext parameter of the according method type and have no return value. You can mark a method with several instances of this attribute.
Examples
Copy
 1/*
 2 vCDL definition:
 3 namespace CalcClient
 4 {
 5   interface ICalculator
 6   {
 7     consumed double Divide(double x, double y, out bool success);
 8   }
 9
10   [Binding="C#"]
11   object Calculator : ICalculator;
12 }
13 */
14[OnTransmitCall(CalcClient.Calculator.MemberIDs.Divide)]
15public void TransmitDivide(CalcClient.ICalculator_Types.Divide.TransientCallContext callContext)
16{
17  var reqID = callContext.ReqID; // remember the request ID for the return
18  // mClient may be e.g. a client for a gRPC call
19  var task = mClient.Divide(callContext.x, callContext.y);
20  task.ContinueWith(t =>
21  {
22    // When the return has arrived, put the received values into a new call context
23    // with the saved request ID and call ReturnReceived
24    Binding.CreateConsumerCallContext(CalcClient.Calculator.Divide, reqID, 
25      out CalcClient.ICalculator_Types.Divide.BindingCallContext callContext2);
26    using (callContext2)
27    {
28      if (!t.IsFaulted)
29      {
30        callContext2.Result = t.Result;
31        callContext2.success = true;
32      }
33      else
34      {
35        callContext2.Result = 0;
36        callContext2.success = false;
37      }
38      Binding.ReturnReceived(CalcClient.Calculator.Divide, callContext2);
39    }
40  });
41}
