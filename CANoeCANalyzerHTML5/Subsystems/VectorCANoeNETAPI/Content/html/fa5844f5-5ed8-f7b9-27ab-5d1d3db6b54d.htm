 1public void VT7009Examples()
 2{
 3  // Get the VTSystem instance
 4  IVTSystem vts = VTSystem.Instance;

 6  // Get the VT7900 module
 7  var vt7900 = vts.GetModule<IVT7900>("VT7900");
 8
 9  // List the available readable values
10  Vector.Tools.Output.WriteLine("Number of writable values: " + vt7900.UserWriteValues.Count.ToString(CultureInfo.InvariantCulture));
11  foreach (string name in vt7900.UserReadValueNames)
12    Vector.Tools.Output.WriteLine(" -> Readable value name: " + name);
13
14  // List the available writable values
15  Vector.Tools.Output.WriteLine("Number of readable values: " + vt7900.UserReadValues.Count.ToString(CultureInfo.InvariantCulture));
16  foreach (string name in vt7900.UserReadValueNames)
17    Vector.Tools.Output.WriteLine(" -> Writable value name: " + name);
18
19  // Turn on the LEDs on the VT7900 front panel for channel 1
20  var channel1 = vts.GetChannel<IVT7900Channel>("VT7900_Ch1");
21  var ledRed = channel1.GetUserValue("LEDRed") as IReadWriteValue<bool>;
22  var ledGreen = channel1.GetUserValue("LEDGreen") as IReadWriteValue<bool>;
23  ledRed.Value = true;
24  ledGreen.Value = true;
25
26  // Turn off the LEDs using direct access to the values from the module
27  vt7900.GetUserWriteValue<bool>("VT7900_Ch1", "LEDRed").Value = false;
28  vt7900.GetUserWriteValue<bool>("VT7900_Ch1", "LEDGreen").Value = false;
29
30  // Read a measurement value from channel 1 and output it to the write window
31  var diffVoltage = channel1.GetUserValue("AInDiff") as IReadValue<double>;
32  if (diffVoltage.IsAvailable)
33    Vector.Tools.Output.WriteLine(diffVoltage.Value.ToString(CultureInfo.InvariantCulture));
34}
