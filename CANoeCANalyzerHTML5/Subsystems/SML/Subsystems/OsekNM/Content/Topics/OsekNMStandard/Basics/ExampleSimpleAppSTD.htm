Example - Simple Application

 

Example

The CAPL code for handling node network management functionality is itemized below.

/*@@var:*/
variables{
  int busActive;                // Transmission of APP-Msg is allowed.
  int SendAppMsg;               // Flag indicates if App-msg should be sent
  int SleepInd;                 // APP is ready to sleep
  int SleepMode;                // APP is sleeping.
  int NMon;                     // Network management is active.
  dword AktualConfig=0;         // current net configuration.

  int WriteRingReq;
  byte RingData[6];

  msTimer timWind;
  const MsgTime=100;
  message apWind apMsg;
}
/*@@end*/

/*@@caplFunc:apBusSleep():*/
apBusSleep ()
{
  // normally, set CAN-Chip to OFF-State.
  write("WIND: apBusSleep called.");
  SleepMode = 1;
  putValue(EnvWindSleepMode,1);
}
/*@@end*/

/*@@caplFunc:apCanSleep():*/
apCanSleep ()
{
  // normally, set CAN-Chip to Standby-State.
  write("WIND: apCanSleep called.");
}
/*@@end*/

/*@@caplFunc:apCanNormal():*/
apCanNormal ()
{
  write("WIND: apCanNormal called.");
  SleepMode = 0; putValue(EnvWindSleepMode,0);
}
/*@@end*/

/*@@caplFunc:apCanWakeUp():*/
apCanWakeUp ()
{
  // don't wake-up, if NM is off.
  if (NMon==0) return;

  write("WIND: apCanWakeUp called.");

  // IMPORTANT: Wake up NM.
  GotoMode_Awake();

  // If APP internal off, set Sleepmode-Indication.
  if (SleepInd) GotoMode_BusSleep();
}
/*@@end*/

/*@@caplFunc:apCanOff():*/
apCanOff ()
{
  // Disable APP-Msg. transmission
  // Check this flag, before any output()-call was made.
  busActive = 0;

  // New CAPL function since CANoe V2.2.
  // This omits all message-calls from
 // this node when NM is in sleep mode.
  CanOffline();
  write("WIND: apCanOff called.");
}

/*@@end*/

/*@@envVar:EnvWindWakeUp:*/
on envVar EnvWindWakeUp
{
  // We will wake-up NM to get bus access.
  if (NMon==0) return;

  // Button is pressed and Node is in Sleepmode.
  if (getValue(this) && SleepMode)
{

  // Wake up NM.
  GotoMode_Awake();

  // We go in active mode, no sleep indication.
  SleepInd = 0;
  putValue(EnvWindSleepInd,0);
  }
}

/*@@end*/

/*@@envVar:EnvWindSleepInd:*/
on envVar EnvWindSleepInd
{
  // Set Sleep indication.
  //---------------------
  SleepInd = getValue(this);
  SetSleepMode();
}
/*@@end*/

/*@@envVar:EnvWindNMOn:*/
on envVar EnvWindNMOn
{
  // Switch NM On or Off.
  NMon = getValue(this);
  if (NMon!=0)
  {
    // Switch NM On.
    TalkNM();

    // Discard SleepMode.
    SleepInd = 0;
    SleepMode = 0;
    putValue(EnvWindSleepInd,0);
    putValue(EnvWindSleepMode,0);
  }
  else
  {
    // Switch NM Off.
    SilentNM();
  }
}
/*@@end*/

/*@@startStart:Start:*/
on start
{
  // On Start, we set all default-values.
  // Get Env settings from Panel.
  WriteRingReq = 0;

  busActive = 1;
  SendAppMsg= getValue(EnvWindSendAppMsg);

  SleepMode = 0;
  putValue(EnvWindSleepMode,0);

  SleepInd = getValue(EnvWindSleepInd);
  SetSleepMode();

  NMon = getValue(EnvWindNMOn);
  if (NMon==0) SilentNM();

  if (SendAppMsg) {
    setTimer(timWind, MsgTime/3);
  }
}
/*@@end*/

/*@@caplFunc:SetSleepMode():*/
SetSleepMode ()
{
  // Depending on SleepInd, we set the internal State
  // of net management.
  if (SleepInd!=0)
  {
    /* Set Sleep mode */
    GotoMode_BusSleep();
  }
  else
{
    /* Reset Sleep-mode Bit */
    GotoMode_Awake();
  }
}
/*@@end*/

/*@@key:'i':*/
on key 'i'
{
  // Get the current network stations participate on NWM-Ring.
  AktualConfig = GetConfig();
  Write("WIND-Config is %lX",AktualConfig);
}
/*@@end*/

/*@@key:'c':*/
on key 'c'
{
  // Get actual status of network management.
  write("Get Status of WIND: %x",NMGetStatus());
}
/*@@end*/

/*@@key:'d':*/
on key 'd'
{
  // Set desired net configuration and mask.
  NMSetTargetConfig(0x40040);
  NMSetConfigMask(0xffffff);
  Write("SollConfig=0x40040, Mask=0xffffff");
}
/*@@end*/

/*@@key:'j':*/
on key 'j'
{
  // Compare actual and desired network configuration.
  // Writes TRUE if all neccessary stations are on guard,
  // otherwise FALSE.
  if (CmpConfig())
  {
    Write("WIND-Config is TRUE");
  }
  else
  {
    Write("WIND-Config is FALSE");
  }
}
/*@@end*/

/*@@caplFunc:apCanOn():*/
apCanOn ()
{
  // APP-transmission is now allowed.
  busActive = 1;


  // New CAPL function since CANoe V2.2.
  // This re-allows all message-calls from
  // this node after NM awakes.
  CanOnline();
  write("WIND: apCanOn called.");
}
/*@@end*/

/*@@timer:timWind:*/
on timer timWind
{
  apMsg.BYTE(0)++;

  // For CANoe-Version lower 2.2,
  // use _output()-function and comment
  // out function call CanOnline()/CanOffline()
  // in function apCanOn and apCanOff().
// _output(apMsg);
  output(apMsg);

  if (SendAppMsg) {
    setTimer(timWind, MsgTime);
  }
}
/*@@end*/

/*@@caplFunc:_output(message*):*/
// Old fashioned version to omit
// message output when Node in Sleepmode
// Since CANoe V2.2 CanOnline() and CanOffline()
// calls can be used (see apCanOn() and apCanOff() )
  output (message *msg)
{
  if (busActive) output(msg);
}
/*@@end*/

/*@@envVar:EnvWindSendAppMsg:*/
on envVar EnvWindSendAppMsg
{
  SendAppMsg = getValue(this);

  if (SendAppMsg) {
      cancelTimer(timWind);
    setTimer(timWind, MsgTime);
  }
}
/*@@end*/

/*@@caplFunc:apIndRingData():*/
apIndRingData ()
{
  if (ReadRingData(6, RingData))
  {
    putValue(EnvWindReadRing6, RingData[0]);
    putValue(EnvWindReadRing5, RingData[1]);
    putValue(EnvWindReadRing4, RingData[2]);
    putValue(EnvWindReadRing3, RingData[3]);
    putValue(EnvWindReadRing2, RingData[4]);
    putValue(EnvWindReadRing1, RingData[5]);
  }
  if (WriteRingReq)
  {
    WriteRingReq=0;
    RingData[0] = getValue(EnvWindWriteRing6);
    RingData[1] = getValue(EnvWindWriteRing5);
    RingData[2] = getValue(EnvWindWriteRing4);
    RingData[3] = getValue(EnvWindWriteRing3);
    RingData[4] = getValue(EnvWindWriteRing2);
    RingData[5] = getValue(EnvWindWriteRing1);

    TransmitRingData(6, RingData);
  }
}
/*@@end*/

/*@@envVar:EnvWindSendRing:*/
on envVar EnvWindSendRing
{
  if (getValue(this))
  {
    WriteRingReq=1;
  }
} /*@@end*/
