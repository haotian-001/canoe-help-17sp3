INI Files - Default INI Content

 

Default AUTOSARFlexrayNM30.INI content

; AUTOSAR FlexRay Network Management configuration file
;
; This file specifies NM parameters for one FlexRay network
; and additional information on individual nodes.
; It is read by the CANoe node layer DLL, which uses the
; network name as base for the INI file name.

[CLUSTER DEFINITIONS]
; Here parameters for the whole cluster, i.e. all NM nodes,
; are specified

; This is the base cycle for the NM protocol, in FlexRay cycles
RepetitionCycleCnt=64

; These values are counts of the repetition cycle
RemoteSleepCnt=2
ReadySleepCnt=5
RepeatMessageCnt=2

; If non-zero, the execution of the FrNm_Main function crosses
; the FlexRay cycle boundary
MainAcrossFrCycle=0

; Location of the NM id within the user data message
NM_PDU_NID_POSITION=1

; Location of the Control Bit Vector in the user data message
NM_PDU_CBV_POSITION=0

; If non-zero, the CBV is copied into the NM vote (schedule 7)
CopyCBVtoNMVote=0

; Here the node specific parameters are defined

;[Node_A]
;NodeIdentifier=0x11
;Passive=0

;[Node_B]
;NodeIdentifier=0x22
;Passive=0

;[Node_C]
;NodeIdentifier=0x33
;Passive=1
