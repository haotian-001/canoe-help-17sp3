Example

The following code illustrates the usage of this DLL in a CAPL simulation node:

includes
{
#include "SetASRTP.cin"
}

variables
{
  // ==========================================================
  // TP addresses to be assigned to the Gateway/Tester and to the ECUs:

  const word cTPAddressGateway = 0xAAA1;
  const word cTPAddressACC2 = 0xBBB1;

  // ==========================================================
  // Pool1 PDUs to be used for diagnostics requests from Gateway/Tester to all ECUs:

  const int cNumPool1PDUs = 16;
  char gPool1PDUArray[cNumPool1PDUs][cPDUNameLength] =

    {
      "ISO_Tester_01", // Length = 12 Bytes
      "ISO_Tester_02", // Length = 12 Bytes
      "ISO_Tester_03", // Length = 12 Bytes
      "ISO_Tester_04", // Length = 12 Bytes
      "ISO_Tester_05", // Length = 12 Bytes
      "ISO_Tester_06", // Length = 12 Bytes
      "ISO_Tester_07", // Length = 12 Bytes
      "ISO_Tester_08", // Length = 12 Bytes
      "ISO_Tester_09", // Length = 12 Bytes
      "ISO_Tester_10", // Length = 12 Bytes
      "ISO_Tester_11", // Length = 12 Bytes
      "ISO_Tester_12", // Length = 12 Bytes
      "ISO_Tester_13", // Length = 12 Bytes
      "ISO_Tester_14", // Length = 12 Bytes
      "ISO_Tester_15", // Length = 12 Bytes
      "ISO_Tester_16" // Length = 12 Bytes
    };
  int gPool1PDULength = 12; // [Bytes]

  // ==========================================================
  // Pool 2 PDUs to be used for diagnostics/flashing responses from ACC2:

  const int cNumPool2PDUs = 2;
  char gPool2PDUArray[cNumPool2PDUs][cPDUNameLength] =
    {
      "ISO_ACC_02_01_Resp", // Length = 12 Bytes
      "ISO_ACC_02_02_Resp" // Length = 12 Bytes
    };
  int gPool2PDULength = 12; // [Bytes]

  // ==========================================================
  // Pool 3 PDUs to be used for flashing requests from Tester/Gateway to all ECUs:

  const int cNumPool3PDUs = 6;
  char gPool3PDUArray[cNumPool3PDUs][cPDUNameLength] =
    {
      "ISO_Flash_01", // Length = 246 Bytes
      "ISO_Flash_02", // Length = 246 Bytes
      "ISO_Flash_03", // Length = 246 Bytes
      "ISO_Flash_04", // Length = 246 Bytes
      "ISO_Flash_05", // Length = 246 Bytes
      "ISO_Flash_06" // Length = 246 Bytes
    };
  int gPool3PDULength = 246; // [Bytes]

  // ==========================================================
  // Pool 4 PDUs to be used for diagnostics/flashing responses from ESP:

  const int cNumPool4PDUs = 2;
  char gPool4PDUArray[cNumPool4PDUs][cPDUNameLength] =
    {
      "ISO_ESP_01_Resp", // Length = 12 Bytes
      "ISO_ESP_02_Resp" // Length = 12 Bytes
    };
  int gPool4PDULength = 12; // [Bytes]

  // ==========================================================
  long gConnectionHandle = 0;
}

/* ========================================================= */

  FrPDU ISO_Tester_07 pdu_1_07;
  FrPDU ISO_Tester_08 pdu_1_08;
  FrPDU ISO_Tester_09 pdu_1_09;
  FrPDU ISO_Tester_10 pdu_1_10;
  FrPDU ISO_Tester_11 pdu_1_11;
  FrPDU ISO_Tester_12 pdu_1_12;
  FrPDU ISO_Tester_13 pdu_1_13;
  FrPDU ISO_Tester_14 pdu_1_14;
  FrPDU ISO_Tester_15 pdu_1_15;
  FrPDU ISO_Tester_16 pdu_1_16;
  FrPDU ISO_Flash_01 pdu_2_01;
  FrPDU ISO_Flash_02 pdu_2_02;
  FrPDU ISO_Flash_03 pdu_2_03;
  FrPDU ISO_Flash_04 pdu_2_04;
  FrPDU ISO_Flash_05 pdu_2_05;
  FrPDU ISO_Flash_06 pdu_2_06;

  // Thus, reserve Tx buffers in CANoe:
  FrSetSendPDU(pdu_1_01);

  FrSetSendPDU(pdu_2_06);
}

/* ========================================================= */
on start
{
  InitTP(%CHANNEL%);
  CreateTPConnection_GW_ACC2(0);
}

/* ========================================================= */
CreateTPConnection_GW_ACC2 (int verbosityLevel)
{

  // Bi-directional diagnostics channel from Gateway to ACC2:

  struct ASRTPParameter params = {
    mProtocolVersion        = eUsedVersion,
    mConnectionModeTx       = eUnicast,
    mConnectionModeRx       = eUnicast,
    mLengthModeTx           = eISO,
    mLengthModeRx           = eISO,
    mAddrType               = 2,
    mLocalAddr              = 0x0000,
    mRemoteAddr             = 0x0000,
    mPadding                = eOn,
    mFillByte               = 0x00,
    mMaxTxPrioPDUs          = 0,
    mBlockSize              = 0,
    mStMin                  = 0,
    mTxPriority             = cDefaultTxPriority,
    mExclusiveTxUsage       = eOff,
    mAbortRxOnNewTxStart    = eOn,
    mMaxPDULength           = 254,
    mTimeoutAs              = gkTimeoutAS,
    mTimeoutAr              = gkTimeoutAR,
    mTimeoutBs              = gkTimeoutBS,
    mTimeBr                 = gkTimeBR,
    mTimeCs                 = gkTimeCS,
    mTimeoutCr              = gkTimeoutCR
  };

  params.mLocalAddr         = cTPAddressGateway;
  params.mRemoteAddr        = cTPAddressACC2;
  params.mMaxPDULength      = gPool1PDULength;
  params.mMaxTxPrioPDUs     = 1;

  CreateTPConnection(cNumPool1PDUs, gPool1PDUArray,
    cNumPool2PDUs, gPool2PDUArray, cPDUNameLength, params,
    verbosityLevel);

  gConnectionHandle = params.mConnectionHandle;
}

/* ========================================================= */
FrTP_ReceptionInd(long connectionHdl, BYTE data[])
{
  // This is a callback from the DLL!
  dword count;
  if (connectionHdl == gConnectionHandle)
  {
    // We are receiving data ...
    count = elcount(data);
    // ...
  }
}

/* ========================================================= */
void TPTransmit (byte buffer[], dword bufferSize)
{
  // We want to send ...
  if (0 == FrTP_DataRequest(gConnectionHandle, buffer, bufferSize))
  {
    SetTPPrefixString(gConnectionHandle);
    write("%sData are sent successful: %d Byte payload",
      gTPPrefix, bufferSize);
  }
else
  {
    SetTPPrefixString(gConnectionHandle);
    write("%sFAILURE! Data sending was not successful: %d Byte",
     gTPPrefix, bufferSize);
  }
}
