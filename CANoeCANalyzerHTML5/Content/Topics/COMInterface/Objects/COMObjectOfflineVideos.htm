Example

Set Application = CreateObject( "CANoe.Application")
' get the path of all offline videos
Set offlineVideos = Application.Configuration.OfflineSetup.OfflineSourceRoot.GetVideoSources
for i = 1 to offlineVideos.Count
  MsgBox "Video " + CStr(i) + ": " + offlineVideos(i).Path
next
' add a video as offline source, and assign it to Video Window 1
Set videoWindow = Application.Configuration.OfflineSetup.VideoWindows(1)
videoSources.Add "C:\MyVideo.avi", videoWindow
