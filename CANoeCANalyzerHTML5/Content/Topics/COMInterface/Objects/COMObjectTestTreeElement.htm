TestTreeElement

Application » Configuration » TestConfigurations » TestConfiguration » TestTreeElements » TestTreeElement

Description

The TestTreeElement object represents an element in the test execution tree. It is used for navigation through the test execution tree.

Example

Example — General

This example traverses a test tree and prints the element information:

Dim CanoeApp, TestConfig, Element, types, verdicts

Set CanoeApp   = CreateObject("CANoe.Application")
Set TestConfig = CanoeApp.Configuration.TestConfigurations(1)

types    = Array(" ","TCF","TU ","TG ","TS ","TC ","TF ","TCL","TSL")
verdicts = Array("n.a.", "pass", "fail")

Call TraverseTree(TestConfig, 0)

Sub TraverseTree(element, level)
  Dim child
  Call PrintElement(element, level)
  For Each child in element.Elements
    Call TraverseTree(child, level + 1)
  Next
End Sub

Sub PrintElement(element, level)
  Dim enabled, indent

  If element.Enabled Then enabled = "*" Else enabled = " "
  indent = String(level * 3, " ")

  WScript.Echo _
    types(element.Type) & " " & _
    enabled & " " & _
    verdicts(element.Verdict) & " " & _
    indent & _
    element.Caption
End Sub

Example — ContainsId method

Set App        = CreateObject("CANoe.Application")
Set TestConfig = App.Configuration.TestConfigurations(1)

' First deactivate all elements
For Each tu in TestConfig.Elements
  tu.Enabled = false
Next

Call EnableElementsWithId(TestConfig, "859e0d2a-5a39-b670-36e4-d320e663070f")
Call EnableElementsWithId(TestConfig, "724d5ae8-d7b9-70c0-7b84-a56d1dfac91a")

Sub EnableElementsWithId(element, testcaseId)
  Dim child
  if element.ContainsId(testcaseId) then
    ' Act element is a testsequence. Containing test cases are not visible in tree.
    ' Enable this element.
    if element.Type = 4 then
      element.Enabled = true
    else
      For Each child in element.Elements
        Call EnableElementsWithId(child, testcaseId)
      Next
    end if
  else
    ' act element is a test case
    if element.Type = 5 and element.Id = testcaseId then
      element.Enabled = true
    end if
  end if
End Sub
