TestUnit

Application » Configuration » TestConfigurations » TestConfiguration » TestUnits » TestUnit

Description

The TestUnit object represents a single test unit within a test configuration.

Example

Example — General

This example displays all test configurations and test units available in a CANoe configuration:

Option Explicit

Dim CANoeApp, TestConfig
Dim tu, enableds, verdicts

enableds = Array("OFF", "ON")
verdicts = Array("n.a.", "Passed", "Failed")

Set CanoeApp = CreateObject("CANoe.Application")

For Each TestConfig in CANoeApp.Configuration.TestConfigurations
  WScript.Echo TestConfig.Name
  For Each tu in TestConfig.TestUnits
    WScript.Echo " ", tu.Name, _
      enableds(Abs(tu.Enabled)), _
      verdicts(tu.Verdict)
  Next
Next

Sub PrintElement(element)
  WScript.Echo _
  "Caption  = " & element.Caption & vbNewLine & _
  "Enabled  = " & element.Enabled & vbNewLine & _
  "Type     = " & element.Type & vbNewLine & _
  "Children = " & element.Elements.Count
End Sub

Example — Caption/Elements/Enabled/Id/Type property

Set App        = CreateObject("CANoe.Application")
Set TestConfig = App.Configuration.TestConfigurations(1)
Set tu         = App.Configuration.TestConfigurations.Item(1).TestUnits.Item(1)

Call TraverseTree(tu)

Sub PrintElement(element)
  Call GetTypeString(element.Type)
  WScript.Echo _
  "Caption  = " & element.Caption & vbNewLine & _
  "Id       = " & element.Id & vbNewLine & _
  "Enabled  = " & element.Enabled & vbNewLine & _
  "Type     = " & TypeString & vbNewLine & _
  "Children = " & element.Elements.Count
End Sub

Sub TraverseTree(element)
  Dim child
  Call PrintElement(element)
  For Each child in element.Elements
    Call TraverseTree(child)
  Next
End Sub

Sub GetTypeString(elementType)
  if elementType = 0 then TypeString = "Reserved" end if
  if elementType = 1 then TypeString = "TestConfiguration" end if
  if elementType = 2 then TypeString = "TestUnit" end if
  if elementType = 3 then TypeString = "TestGroup" end if
  if elementType = 4 then TypeString = "TestSequence" end if
  if elementType = 5 then TypeString = "TestCase" end if
  if elementType = 6 then TypeString = "TestFixture" end if
  if elementType = 7 then TypeString = "TestCaseList" end if
  if elementType = 8 then TypeString = "TestSequenceList" end if
End Sub

Example — Report property

This example starts a CANoe test configuration, waits for the report generation and opens the HTML report with the default browser when the report generation is finished:

Option Explicit

Dim WshShell, CanoeApp, Configuration, Measurement
Dim TestConfiguration, TestConfigurationReport
Dim TestUnit, TestUnitReport
Dim measuring, testing, reported

WScript.Quit(Main)

Function Main
  Set WshShell                = WScript.CreateObject("WScript.Shell")
  Set CanoeApp                = CreateObject("CANoe.Application")
  Set Configuration           = CanoeApp.Configuration
  Set Measurement             = CanoeApp.Measurement
  Set TestConfiguration       = Configuration.TestConfigurations.Item(1)
  Set TestConfigurationReport = TestConfiguration.Report
  Set TestUnit                = TestConfiguration.TestUnits(1)
  Set TestUnitReport          = TestUnit.Report

  WScript.ConnectObject Measurement, "Measurement_"
  WScript.ConnectObject TestConfiguration, "TestConfiguration_"
  If TestConfigurationReport.UseJointReport Then
    WScript.ConnectObject TestConfigurationReport, "TestConfigurationReport_"
  Else
    If TestUnitReport.Enabled Then
      WScript.ConnectObject TestUnitReport, "TestUnitReport_"
    Else
      ' reporting is disabled
      WScript.Echo "HTML reporting is disabled."
      Main = -1
      Exit Function
    End If
  End If

  ' start measurement
  measuring = Measurement.Running
  If Not measuring Then
    Measurement.Start
  End If

  ' wait for measurement start
  While Not measuring
    Wscript.Sleep 100
  Wend

  testing = False
  ' start testing
  TestConfiguration.Start

  ' wait for testing start or measurement end
  While Not testing and measuring
    Wscript.Sleep 100
  Wend

  ' wait for reporting event or measurement end
  reported = False
  While Not reported and measuring
    Wscript.Sleep 100
  Wend
    Main = 0
End Function

Sub Measurement_OnStart()
  measuring = True
End Sub

Sub Measurement_OnStop()
  measuring = False
End Sub

Sub TestConfiguration_OnStart()
  testing = True
End Sub

Sub TestConfiguration_OnStop(reason)
  testing = False
End Sub

Sub TestUnitReport_OnGenerated(success, sourceFullPath, generatedFullPath)
  reported = True
  If success Then
    OpenReport(generatedFullPath)
  Else
    WScript.Echo "Test unit HTML report failed."
  End If
End Sub

Sub TestConfigurationReport_OnGenerated(success, sourceFullPath, generatedFullPath)
  reported = True
  If success Then
    OpenReport(generatedFullPath)
  Else
    WScript.Echo "Test configuration HTML report failed."
  End If
End Sub

Sub OpenReport(path)
  WshShell.Run """" & path & """"
End Sub

Example — ContainsId method

Set App        = CreateObject("CANoe.Application")
Set TestConfig = App.Configuration.TestConfigurations(1)

' First deactivate all elements
For Each tu in TestConfig.Elements
  tu.Enabled = false
Next

Call EnableElementsWithId(TestConfig, "859e0d2a-5a39-b670-36e4-d320e663070f")
Call EnableElementsWithId(TestConfig, "724d5ae8-d7b9-70c0-7b84-a56d1dfac91a")

Sub EnableElementsWithId(element, testcaseId)
  Dim child
  if element.ContainsId(testcaseId) then
    ' Act element is a testsequence. Containing test cases are not visible in tree.
    ' Enable this element.
    if element.Type = 4 then
      element.Enabled = true
    else
      For Each child in element.Elements
        Call EnableElementsWithId(child, testcaseId)
      Next
    end if
  else
    ' act element is a test case
    if element.Type = 5 and element.Id = testcaseId then
      element.Enabled = true
    end if
  end if
End Sub

Example — GetVariant / SetVariant method

This example reads the current variants for Country and Coverage and sets new variant values:

Dim CanoeApp, TestConfig, TestUnit, value
Dim country, coverage

Set CanoeApp   = CreateObject("CANoe.Application")
Set TestConfig = CanoeApp.Configuration.TestConfigurations(1)
Set TestUnit   = TestConfig.TestUnits("CentralLockingSystemTest_CSharp")

country  = TestUnit.GetVariant("Country")
coverage = TestUnit.GetVariant("Coverage")

WScript.Echo "Current variant = " & country, coverage

' set the variant property using value table entries
TestUnit.SetVariant "Country", "Europe"
TestUnit.SetVariant "Coverage", "Full"

TestConfig.ApplyVariants

country  = TestUnit.GetVariant("Country")
coverage = TestUnit.GetVariant("Coverage")

WScript.Echo "New variant = " & country, coverage
