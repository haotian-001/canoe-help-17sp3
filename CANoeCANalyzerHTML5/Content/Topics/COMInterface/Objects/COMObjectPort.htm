Description

The Port object represents a specific port of a CANoe configuration.

Ports are the access points for applications on the network. They can be used for simulation (read/write access) or for measurement (read access).

Example

The following example prints all available ports of all networks for CANoe:

Set App             = CreateObject("CANoe.Application")
Set Configuration   = App.Configuration
Set SimulationSetup = Configuration.SimulationSetup
Set Buses           = SimulationSetup.Buses

Wscript.Echo "Configuration: " & Configuration.Name

ListPorts()

Sub ListPorts()
   ' List configuration of simulation nodes
  for BusIndex = 1 to SimulationSetup.Buses.Count
    Set Bus   = Buses.Item(BusIndex)
    Set Ports = Bus.Ports

    if Ports.IsSwitchedNetwork then
      Wscript.Echo ""
      Wscript.Echo "###################################################"
      Wscript.Echo "Network: " & Ports.NetworkName
      PrintPorts Ports
    End if
  next
End Sub

Sub PrintPorts(portList)
  for PortIndex = 1 to portList.Count
    Set Port = portList.Item(PortIndex)
    PrintPort Port
  next
End Sub

Sub PrintPort(port)
  Wscript.Echo "-------------- Port " & port.Name & " -------------------"
  Wscript.Echo "Segment Name: " & port.SegmentName
  Wscript.Echo "Is Simulation Port: " & CSTR(port.IsSimulated)
  Wscript.Echo "Is Active: " & CSTR(port.IsActive)
End Sub
