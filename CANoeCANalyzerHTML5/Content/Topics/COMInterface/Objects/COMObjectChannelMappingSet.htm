Description

The ChannelMappingSet object allows setting or returning the channel mapping configuration for the offline source.

Example

Example — ClearMappingTable method

Clear the default mapping:

Set App = CreateObject( "CANoe.Application")
Set defaultMapping = App.Configuration.OfflineSetup.OfflineSourceRoot.GetMappingSet
if not defaultMapping is Nothing then
  defaultMapping.ClearMappingTable
end if

Example — Copy method

' Create a copy of the default mapping, or create a new one
Set App = CreateObject( "CANoe.Application")
Set Setup = App.Configuration.OfflineSetup
Set defaultMapping = Setup.OfflineSourceRoot.GetMappingSet
if defaultMapping is Nothing then
  Set newMapping = Setup.CreateMappingSet("mapping")
else
  Set newMapping = defaultMapping.Copy
end if
' Modify the mapping: ignore CAN 1
Const cCan = 1
newMapping.Put cCan, 1, 0
' Assign the mapping to the first logfile
Setup.OfflineSourceRoot.GetLogFiles(1).SetMappingSet(newMapping)

Example — Get method

' Get the default channel mapping of CAN 1 to 4
Const cCAN = 1
Set App = CreateObject( "CANoe.Application")
Set defaultMapping = App.Configuration.OfflineSetup.OfflineSourceRoot.GetMappingSet
if defaultMapping is Nothing then
  MsgBox "Default mapping set is not set"
else
  for src = 1 to 4
    dst = defaultMapping.Get(cCAN, src)
    MsgBox "CAN " + CStr(src) + " -> " + CStr(dst)
  next
end if

Example — GetByBusName method

Get the default channel mapping of CAN 1 to 4:

Set App = CreateObject( "CANoe.Application")
Set defaultMapping = App.Configuration.OfflineSetup.OfflineSourceRoot.GetMappingSet
if defaultMapping is Nothing then
  MsgBox "Default mapping set is not set"
else
  for src = 1 to 4
    dst = defaultMapping.GetByBusName("CAN", src)
    MsgBox "CAN " + CStr(src) + " -> " + CStr(dst)
  next
end if

Example — Put method

' Create a channel mapping set that swaps CAN 1 and CAN 2
Const cCAN = 1
Set App = CreateObject( "CANoe.Application")
Set mapping = App.Configuration.OfflineSetup.CreateMappingSet("swap")
mapping.Put cCAN, 1, 2
mapping.Put cCAN, 2, 1' Create a channel mapping set that swaps CAN 1 and CAN 2
