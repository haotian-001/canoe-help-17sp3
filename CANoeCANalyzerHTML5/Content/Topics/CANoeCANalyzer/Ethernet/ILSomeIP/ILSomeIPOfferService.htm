SOME/IP IL: Offer a Service

 

You can use the SOME/IP Interaction Layer to create a simulation node that offers a Service. To do this, you must assign the SOME/IP Interaction Layer to this node. Then you can configure the service in the CAPL node of the network node.

variables
{
  char gMulticastIPStr[16] = "239.192.255.251";
  dword gMulticastIP;
  dword gProvidedEventHandle;
  LONG gClientID           = 2;
  dword kUDP               = 17;
  Timer gTimer;
  dword gNumber;
}

on start
{
  dword appEndpointHandle;
  dword serviceHandle;
  dword providedEventgroupHandle;

  gMulticastIP = IpGetAddressAsNumber(gMulticastIPStr);
  appEndpointHandle = SomeIpOpenLocalApplicationEndpoint(kUDP, 30490);

  //create a provided service instance
  serviceHandle = SomeIpCreateProvidedServiceInstance(appEndpointHandle, 257 /*serviceId*/, 54 /*instanceId*/);
  SomeIpSetProperty(serviceHandle, "SDCyclicOfferDelay", 1000);  //service is announced every second
  SomeIpSetProperty(serviceHandle, "SDTTL", 3);  //service offer is valid 3 seconds

  //creates a provided event group
  providedEventgroupHandle = SomeIpAddProvidedEventGroup(serviceHandle, 1 /*eventgroupId*/);
  SomeIpSetProperty(providedEventgroupHandle, "MulticastIp" , gMulticastIP);  //events are sent per multicast
  SomeIpSetProperty(providedEventgroupHandle, "MulticastPort", 30490);

  //create provided event
  gProvidedEventHandle = SomeIpAddEvent(serviceHandle, 32779/*eventId*/, "OnPrepareEvent");
  SomeIpAddEventToEventgroup(providedEventgroupHandle, gProvidedEventHandle);
  setTimer(gTimer, 1);
}

on timer gTimer
{
  //event is sent cyclically
  SomeIpTriggerEvent(gProvidedEventHandle);
  setTimer(gTimer, 1);
  gNumber++;
}

void OnPrepareEvent(dword eventHandle, dword messageHandle)
{
  // "signal" must be a valid path according to the FIBEX database that is configured for the current bus
  SomeIpSetValueLong(messageHandle, "signal", gNumber);
}
