In XML test modules, you can modify these settings using the vtsystem_configure command.

Example Test Module

CAPL

// Configure output to "high side switch"
sysvar::VTS::DoorSwitch.SetOutputMode (eVTS2848OutputModeHighsideSwitch);

// Start stimulating "DoorSwitch" channel
sysvar::VTS::DoorSwitch.StartStimulation ();

// Configure source for high output to "Vext" for channels 1 - 4
sysvar::VTS::DoorIOs.SetOutputSource (eVTSOutputSourceGroupChannels1To4, eVTSOutputSourceVExt);

// Set 5.0 V as threshold for channels 1-8 of "DoorIOs" module
sysvar::VTS::DoorIOs.SetThreshold (0, 5.0);

.NET (#C)

IVTSystem vts = VTSystem.Instance;

// Get the module "DoorIOs"
IVT2848 doorsIO = vts.GetModule<IVT2848>("DoorIOs");

// Configure source for high output to "Vext" for channels 1 - 4
doorsIO.OutputSource1To4.Value = OutputSource.VExt;

// Set 5.0 V as threshold for channels 1-8 of "DoorIOs" module
doorsIO.Threshold1To8.Value = 5.0;

// Get the channel "DoorSwitch"
IVT2848PWMStimChannel doorSwitch = vts.GetChannel<IVT2848PWMStimChannel>("DoorSwitch");

// Configure output to "high side switch"
doorSwitch.OutputMode.Value = VT2848OutputMode.HighsideSwitch;

// Start stimulating "DoorSwitch" channel
doorSwitch.StartStimulation();
