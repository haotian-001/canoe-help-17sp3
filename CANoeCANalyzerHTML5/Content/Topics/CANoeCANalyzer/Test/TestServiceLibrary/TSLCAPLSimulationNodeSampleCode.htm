Test Service Library: Sample Code (CAPL Simulation Node)

 

variables
{
  dword mCheckId = 0;                 // Remember the id of the currently running check
  const int cTextBufferLength = 512;  // For the textual status query
  dword mMinDist = 1000;              // Initial minimal distance between the
                                      // occurrence of Ref-Msg and Obs-Msg in ms
  dword mMaxDist = 6000;              // Initial maximal distance between the
                                      // occurrence of Ref-Msg and Obs-Msg in ms
}

on preStart
{
  ChkConfig_Init ("user");  // Initialize the test functionality
  // Now the concrete checks are setup (and commonly started)
  // The connected database contains the following messages:
  // «RefMsg» and «ObsMsg»
  SetupCheck();
}

Callback (dword aCheckId)
{
  long lEventMsgId = 0;
  char lEventMsgName[cTextBufferLength];
  long lQueryResultOnMsgName = 0;
  long lTime = 0;
  long lStatNumEvents = 0;
  lQueryResultOnMsgName = ChkQuery_EventMessageName (aCheckId, lEventMsgName, cTextBufferLength);
  lEventMsgId = ChkQuery_EventMessageId (aCheckId);
  lTime = ChkQuery_EventInterval (aCheckId);

  if(lTime < 0)
  {
    // In this case, the message distance was too big.
    // Refer the error codes of this query
    // Let's resetup the check more liberal on the max distance
    mMaxDist = mMaxDist * 1.1;
  }
  else
  {
    // In the case, the message distance was too short.
    // Let's resetup the check more liberal on the min distance
    mMinDist = mMinDist * 0.9;
  }
  write("Callback for MsgDist: Msg %s(%d) had interval %d ", lEventMsgName,
  lEventMsgId,
  lTime);
  lStatNumEvents  = ChkQuery_NumEvents (aCheckId);
  write("Event-Statistics: Count=%d", lStatNumEvents);
  // Now we destroy the check and do setup it more liberal
  ChkControl_Destroy (mCheckId);
  SetupCheck()
}

SetupCheck ()
{
  mCheckId = ChkCreate_MsgDistViolation("RefMsg",  // Reference message
  "ObsMsg",     // observed message
  mMinDist,     // Minimum distance in milliseconds
  mMaxDist,     // Maximum distance in milliseconds
  "Callback");  // CAPL-Callback-function
  write("We have setup the check with id %d with min=%u and max=%u", mCheckId, mMinDist, mMaxDist);
}
